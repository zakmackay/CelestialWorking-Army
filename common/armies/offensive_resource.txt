hover_tank_army = {
	damage = 2
	health = 2
	morale = 1
	morale_damage = 0.75
	time = 180
	maintenance = 0.4
	icon_frame = 9
	
	cost = {
		minerals = 100
		energy = 40
	}
	
	prerequisites = {  }

	allow = {
		custom_tooltip = {
		text = force_limit_spec
			owner = {
				check_variable = {
					which = "hover_tank_count" 
					value > 0
				}
			}
		}
		custom_tooltip = {
			text = force_limit
			owner = {
				check_variable = {
					which = "force_limit" 
					value > 0
				}
			}
		}
	}
	on_queued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = -3
			}
			change_variable = {
				which = "hover_tank_count" 
				value = -3
			}
		}
	}

	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = 3
			}
			change_variable = {
				which = "hover_tank_count" 
				value = 3
			}
		}
	}
}
