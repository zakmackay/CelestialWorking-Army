xeno_cav_psi_army = {
	damage = 2.5
	morale_damage = 2
	health = 1.5
	morale = 2
	time = 180
	maintenance = 0.4
	icon_frame = 9
	
	cost = {
		minerals = 150
		energy = 50
	}
	
	prerequisites = { "tech_telepathy" "tech_gene_expressions" }

	allow = {

		custom_tooltip = {
		text = force_limit_spec
			owner = {
				check_variable = {
					which = "xeno_cav_count" 
					value > 0
				}
			}
		}
		custom_tooltip = {
			text = force_limit
			owner = {
				check_variable = {
					which = "force_limit" 
					value > 0
				}
			}
		}
		custom_tooltip = {
			text = force_limit
			owner = {
				check_variable = {
					which = "force_limit_specs" 
					value > 0
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = -3
			}
			change_variable = {
				which = "force_limit_specs" 
				value = -3
			}
			change_variable = {
				which = "xeno_cav_count" 
				value = -3
			}
		}
	}

	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = 3
			}
			change_variable = {
				which = "force_limit_specs" 
				value = -3
			}
			change_variable = {
				which = "xeno_cav_count" 
				value = 3
			}
		}
	}
}
