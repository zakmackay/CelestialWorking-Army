
# Defense Armies
defense_army = {
	defensive = yes
	time = 60
	health = 1.0
	damage = 1.0
	morale = 1
	icon_frame = 1
	maintenance = 0.15
	
	cost = {
		minerals = 30
	}
	
	prerequisites = { "tech_defense_army" }

	allow = {
		custom_tooltip = {
			text = force_limit
			owner = {
				check_variable = {
					which = "force_limit" 
					value > 0
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = -1
			}
		}
	}
	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = 1
			}
		}
	}
}
