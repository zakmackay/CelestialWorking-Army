
# Synthetic Armies
android_army_def = {
	defensive = yes
	robotic = yes
	damage = 1.6
	health = 2.5
	time = 90
	maintenance = 0.20
	icon_frame = 4
	has_species = no
	
	cost = {
		minerals = 75
		energy = 10
	}
	
	prerequisites = { "tech_synthetic_workers" "tech_defense_army"  }

	allow = {
		custom_tooltip = {
			text = force_limit
			owner = {
				check_variable = {
					which = "force_limit" 
					value > 0
				}
			}
		}
		custom_tooltip = {
			text = force_limit_spec
			owner = {
				check_variable = {
					which = "force_limit_spec" 
					value > 0
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = -1
			}
			change_variable = {
				which = "force_limit_spec" 
				value = -1
			}
		}
	}

	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = 1
			}
			change_variable = {
				which = "force_limit_spec" 
				value = 1
			}
		}
	}
}
