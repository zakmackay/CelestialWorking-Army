
# Synthetic Armies
android_army = {
	robotic = yes
	damage = 1.6
	health = 2.5
	time = 120
	maintenance = 0.4
	icon_frame = 4
	has_species = no
	
	cost = {
		minerals = 150
		energy = 20
	}
	
	prerequisites = { "tech_synthetic_workers" }

	allow = {
		custom_tooltip = {
			text = force_limit
			owner = {
				check_variable = {
					which = "force_limit" 
					value > 0
				}
			}
		}
		custom_tooltip = {
			text = force_limit_spec
			owner = {
				check_variable = {
					which = "force_limit_spec" 
					value > 0
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = -3
			}
			change_variable = {
				which = "force_limit_spec" 
				value = -3
			}
		}
	}

	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = 3
			}
			change_variable = {
				which = "force_limit_spec" 
				value = 3
			}
		}
	}
}
