hover_tank_army_def = {
	defensive = yes
	damage = 2
	morale_damage = 0.75
	health = 2
	morale = 1
	time = 120
	maintenance = 0.2
	icon_frame = 9
	
	cost = {
		minerals = 50
		energy = 20
	}
	
	prerequisites = {  "tech_defense_army" }

	allow = {
		custom_tooltip = {
		text = force_limit_spec
			owner = {
				check_variable = {
					which = "hover_tank_count" 
					value > 0
				}
			}
		}
		custom_tooltip = {
			text = force_limit
			owner = {
				check_variable = {
					which = "force_limit" 
					value > 0
				}
			}
		}
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = -1
			}
			change_variable = {
				which = "hover_tank_count" 
				value = -1
			}
		}
	}

	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "force_limit" 
				value = 1
			}
			change_variable = {
				which = "hover_tank_count" 
				value = 1
			}
		}
	}
}
